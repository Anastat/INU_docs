components/OptionsSwitch.vue

Child component OptionsSwitch.vue Component displays input switch.

Methods:

-> openModalFromParent() - Calls method from parent component Options.vue on switch input to true

<template>
    <div>
        <div class="inputGroup">
            <div class="switch">
                <label class="switch__label" :for="option.id">{{
              option.labelText
            }}</label>
    
                <input class="switch__checkbox" type="checkbox" :id="option.id" :checked="option.checked"
                    @click="openModalFromParent(option.id)" />
                <label class="switch__handle has-bgColor-stateBlue250" :for="option.id">
                    <span class="switch__icon">
                </label>
            </div>
        </div>
    </div>
</template>

<script>
export default {
    name: "OptionsSwitch",
    props: {
        option: {
            type: Object
        }
    },
    methods: {
        openModalFromParent: function(id) {
            this.$parent.openModal(id);
        }
    }
};
</script>
